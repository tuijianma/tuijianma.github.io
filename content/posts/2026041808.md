```python
from binance.client import Client
from binance.enums import *
import pandas as pd
import time

# 1. 用你的API信息替换此处
API_KEY = ‘YOUR_API_KEY_HERE‘
API_SECRET = ‘YOUR_SECRET_KEY_HERE‘

# 2. 初始化客户端，连接到币安合约测试网（主网切换为 `https://fapi.binance.com`）
client = Client(API_KEY, API_SECRET, testnet=True) # 首次测试务必使用测试网！

# 3. 获取账户信息（只读，安全）
def get_account_info():
    try:
        account_info = client.futures_account()
        print(f"账户总权益: {account_info['totalWalletBalance']} USDT")
        print(f"可用保证金: {account_info['availableBalance']} USDT")
        # 打印当前持仓
        positions = [p for p in account_info['positions'] if float(p['positionAmt']) != 0]
        if positions:
            print("\n当前持仓:")
            for pos in positions:
                print(f"  合约: {pos['symbol']}, 数量: {pos['positionAmt']}, 盈亏: {pos['unrealizedProfit']} USDT")
        else:
            print("\n当前无持仓。")
    except Exception as e:
        print(f"获取账户信息失败: {e}")

# 4. 监控BTCUSDT价格
def monitor_price(symbol=‘BTCUSDT‘):
    print(f"\n开始监控 {symbol} 价格...")
    try:
        # 获取最新价格
        ticker = client.futures_symbol_ticker(symbol=symbol)
        print(f"当前价格: {ticker['price']}")
        
        # 获取深度信息
        depth = client.futures_order_book(symbol=symbol, limit=5)
        best_bid = depth['bids'][0][0] # 最优买价
        best_ask = depth['asks'][0][0] # 最优卖价
        print(f"买一价: {best_bid}, 卖一价: {best_ask}, 价差: {float(best_ask)-float(best_bid):.2f}")
        
    except Exception as e:
        print(f"获取价格失败: {e}")

# 执行函数
if __name__ == ‘__main__‘:
    print("=== 币安合约API连接测试 ===")
    get_account_info()
    monitor_price()
```
